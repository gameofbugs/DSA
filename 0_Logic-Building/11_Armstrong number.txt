namespace DSA_Problems;

class Program
{
    static void Main(string[] args)
    {
        Console.WriteLine("Enter a number");
        int n = Convert.ToInt32(Console.ReadLine());
        int num = n;
        int length = 0;
        int armstrongNumber = 0;
        while (num > 0)
        {
            num = num / 10;
            length++;
        }
        num = n;
        while (num > 0)
        {
            int digit = num % 10;
            num = num / 10;
            armstrongNumber += (int)MathF.Pow(digit, length);
        }
        if (n == armstrongNumber)
        {
            Console.WriteLine($"The Number {n} is Armstorng Number");
            Console.WriteLine(armstrongNumber);
        }
        else
        {
            Console.WriteLine($"The Number {n} is not Armstorng Number");
            Console.WriteLine(armstrongNumber);
        }

    }
}
