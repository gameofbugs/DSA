namespace DSA_Problems;

class Program
{
    static void Main(string[] args)
    {
        Console.WriteLine("Enter a Number");
        int n = Convert.ToInt32(Console.ReadLine());
        int rev = 0;
        while (n > 0)
        {
            int digit = n % 10;
            rev = rev * 10 + digit;
            n = n / 10;
        }
        Console.WriteLine(rev);
    }
}
