namespace DSA_Problems;

class Program
{
    static void Main(string[] args)
    {
        Console.WriteLine("Enter a number");
        int n = Convert.ToInt32(Console.ReadLine());

        int prev = 0;
        int current = 1;
        int next = 0;

        for (int i = 0; i <= n; i++)
        {

            Console.WriteLine(prev);
            next = prev + current;
            prev = current;
            current = next;
        }
    }
}
