namespace DSA_Problems;

class Program
{
    static void Main(string[] args)
    {
        int[] array = { 10, 5623, 30, 40, 99 };

        int searchelement = 99;
        int index = -1;
        for (int i = 0; i < array.Length; i++)
        {
            if (array[i] == searchelement)
            {
                index = i;
                break;
            }
        }
        if (index < 0)
        {
            Console.WriteLine($"The element {searchelement} does not exist in an array");
            return;
        }
        else
        {
            Console.WriteLine($"The element {searchelement} found at {index} posotion");
        }
    }
}
