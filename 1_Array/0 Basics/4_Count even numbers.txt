namespace DSA_Problems;

class Program
{
    static void Main(string[] args)
    {
        int[] array = { 10, 5623, 30, 40 };
        int count = 0;
        foreach (int i in array)
        {
            if (i % 2 == 0)
            {
                count++;
            }
        }
        Console.WriteLine(count);
    }
}
