namespace DSA_Problems;

class Program
{
    static void Main(string[] args)
    {
        int[] a = { 1, 2, 3, 1 };
        bool isContainDuplicate = false;

        for (int i = 0; i < a.Length; i++)
        {
            for (int j = i + 1; j < a.Length; j++)
            {
                if (a[i] == a[j])
                {
                    isContainDuplicate = true;
                    break;
                }
            }
        }
        if (isContainDuplicate)
        {
            System.Console.WriteLine($"Yes the given array contain duplicaate elements");
        }
        else
        {
            System.Console.WriteLine($"No the given array contain duplicaate elements");
        }
    }
}
