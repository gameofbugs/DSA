namespace DSA_Problems;

class Program
{
    static void Main(string[] args)
    {
        int[] a = { 5, 2, 1, 3, 4 };
        int target = 9;

        for (int i = 0; i < a.Length; i++)
        {
            for (int j = i + 1; j < a.Length; j++)
            {
                if (a[i] + a[j] == target)
                {
                    System.Console.WriteLine($"{a[i]} and {a[j]}");
                    break;
                }
            }
        }
    }
}
